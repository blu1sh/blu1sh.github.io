    @Override
public iMatrix integrate(iIntegrand integrand, iTopology topology, int order)
{
	final Quadrature q = new Quadrature(order);
	iMatrix ans = integrand.shape();
	iMatrix xi;
	for (int i = 0; i < order; i++)
	{
		for (int j = 0; j < order; j++)
		{
			xi = q.p(i, j);
			ans = MBOs.add(
				ans,MBOs.scale(
					q.w(i, j) * topology.det_J(xi),
					integrand.integrand(xi)
				)
			);
		}
	}
	return ans;
}
